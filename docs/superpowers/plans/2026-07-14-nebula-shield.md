# Nebula Tuning & Ship-Silhouette Shield Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Tune the nebula field (rarer, multi-lobed/irregular, some clouds independent of the star ladder) and replace the plain-circle shield with an outline traced from the player ship's own pixel-art silhouette.

**Architecture:** Nebula changes are almost entirely data-shape changes in `Perihelion.Nebula` (`lib/perihelion_nebula.march`) — `star_cloud` goes from returning one optional cloud to a small list of clustered lobes, and a second, independent free-floating spawn mechanism is added alongside it. The WebGL shader already sums density from an arbitrary list of clouds per pixel, so none of this needs shader changes. The shield outline is a pure-geometry derivation from the same 64x96 pixel grid the pixel-art-ship feature already extracted (`player_sprite_grid()` in `lib/perihelion.march`), pre-computed once at boot (mirroring how the sprite bitmap itself is pre-rendered once) and stroked as a closed path inside `draw_ball`'s existing rotate/translate block.

**Tech Stack:** March (interpreted + compiled JS target), existing `Perihelion.Nebula`/`Perihelion`/`WebGL` stdlib modules — no new dependencies.

- [ ] **Step 2: Turn `star_cloud` into a multi-lobe generator**

Find:

```march
  -- The cloud (if any) anchored at a star's position. Jitter is small
  -- relative to radius_min, so wherever this returns Some, the star always
  -- falls well inside the cloud's body -- a spawned cloud always overlaps
  -- the star it's rooted to.
  pfn star_cloud(sx : Float, sy : Float, seed : Float) : Option(Cloud) do
    if hash3(sx, sy, seed) > cloud_chance() do
      None
    else
      let jx = (hash3(sx + 1.0, sy, seed) - 0.5) * 2.0 * anchor_jitter()
      let jy = (hash3(sx, sy + 1.0, seed) - 0.5) * 2.0 * anchor_jitter()
      -- Square the raw hash before lerping so sizes skew toward the small
      -- end with a real tail of much bigger clouds, instead of a flat
      -- uniform spread -- most clouds modest, a few genuinely huge.
      let rt = hash3(sx + 2.0, sy, seed)
      let r = lerp(radius_min(), radius_max(), rt * rt)
      let strength = 0.35 + 0.65 * hash3(sx, sy + 2.0, seed)
      Some({ x: sx + jx, y: sy + jy, radius: r, strength: strength })
    end
  end
```

Replace with:

```march
  -- How many lobes a triggered star-cloud gets: 2 or 3, chosen by an
  -- independent hash roll so it varies per star/seed like everything else.
  pfn lobe_count(sx : Float, sy : Float, seed : Float) : Int do
    if hash3(sx + 3.0, sy, seed) < 0.5 do 2 else 3 end
  end

  -- One lobe of a star's cloud cluster. Lobe 0 uses the original tight
  -- anchor_jitter() bound, so it always overlaps its star exactly as a
  -- single-lobe cloud used to -- the "a spawned cloud always overlaps the
  -- star it's rooted to" invariant is preserved via this lobe specifically.
  -- Later lobes (1, 2) jitter further (up to half the base radius) so the
  -- cluster reads as one bulging, irregular shape rather than a neat circle,
  -- and may drift far enough that they don't individually overlap the star.
  pfn make_lobe(sx : Float, sy : Float, seed : Float, base_r : Float, i : Int) : Cloud do
    let fi = int_to_float(i)
    let spread = if i == 0 do anchor_jitter() else base_r * 0.5 end
    let jx = (hash3(sx + 10.0 + fi, sy, seed) - 0.5) * 2.0 * spread
    let jy = (hash3(sx, sy + 10.0 + fi, seed) - 0.5) * 2.0 * spread
    let rf = 0.5 + 0.35 * hash3(sx + 20.0 + fi, sy, seed)
    let strength = 0.35 + 0.65 * hash3(sx, sy + 20.0 + fi, seed)
    { x: sx + jx, y: sy + jy, radius: base_r * rf, strength: strength }
  end

  pfn make_lobes(sx : Float, sy : Float, seed : Float, base_r : Float, i : Int, n : Int, acc : List(Cloud)) : List(Cloud) do
    if i >= n do
      acc
    else
      make_lobes(sx, sy, seed, base_r, i + 1, n, Cons(make_lobe(sx, sy, seed, base_r, i), acc))
    end
  end

  -- The cloud lobes (if any) anchored at a star's position -- 0 if the
  -- cloud_chance() roll misses, otherwise 2-3 lobes clustered near the star
  -- (see make_lobe). Empty-list-vs-Option mirrors how visible_clouds already
  -- returns a plain List(Cloud) -- callers never branched on Option here.
  pfn star_cloud(sx : Float, sy : Float, seed : Float) : List(Cloud) do
    if hash3(sx, sy, seed) > cloud_chance() do
      Nil
    else
      -- Square the raw hash before lerping so sizes skew toward the small
      -- end with a real tail of much bigger clouds, instead of a flat
      -- uniform spread -- most clouds modest, a few genuinely huge.
      let rt = hash3(sx + 2.0, sy, seed)
      let base_r = lerp(radius_min(), radius_max(), rt * rt)
      make_lobes(sx, sy, seed, base_r, 0, lobe_count(sx, sy, seed), Nil)
    end
  end
```

```march
  type SpriteRowExtent = { row : Int, left : Int, right : Int }

  -- The (leftmost, rightmost) non-background column in a sprite grid row,
  -- or (-1, -1) if the row is entirely background ("0" in every column).
  pfn row_extent_go(row : String, col : Int, len : Int, left : Int, right : Int) : (Int, Int) do
    if col >= len do
      (left, right)
    else
      let ch = String.slice(row, col, 1)
      if ch == "0" do
        row_extent_go(row, col + 1, len, left, right)
      else
        let new_left = if left < 0 do col else left end
        row_extent_go(row, col + 1, len, new_left, col)
      end
    end
  end

  pfn row_extent(row : String) : (Int, Int) do
    row_extent_go(row, 0, 64, 0 - 1, 0 - 1)
  end

  -- Walks player_sprite_grid()'s 96 rows top-to-bottom, but Cons-prepending
  -- means the returned list comes out BOTTOM-to-TOP (last-processed row is
  -- the head) -- used directly as the left edge's natural walk order in
  -- shield_hull_points below. Rows that are entirely background are
  -- skipped (no left/right to record).
  pfn collect_row_extents(rows : List(String), row_idx : Int, acc : List(SpriteRowExtent)) : List(SpriteRowExtent) do
    match rows do
      Nil -> acc
      Cons(row, rest) ->
        let (l, r) = row_extent(row)
        let acc2 = if l < 0 do acc else Cons({ row: row_idx, left: l, right: r }, acc) end
        collect_row_extents(rest, row_idx + 1, acc2)
    end
  end

  pfn sprite_row_extents() : List(SpriteRowExtent) do
    collect_row_extents(player_sprite_grid(), 0, Nil)
  end

  -- How far the shield hull is scaled out from the sprite's own silhouette,
  -- from its center -- so the shield visibly surrounds the ship rather than
  -- tracing directly on top of it.
  fn shield_scale() : Float do 1.15 end
  -- Grid-space (64 cols x 96 rows) to the sprite's own local pixel-space
  -- (32x48, matching draw_player_sprite_pixels's target size).
  fn sprite_col_scale() : Float do 0.5 end
  fn sprite_row_scale() : Float do 0.5 end
  -- Half the sprite's local pixel-space size -- both its own center point
  -- and, since Canvas.draw_node draws the sprite at (-16, -24), the offset
  -- needed so hull points land in the same local frame as the sprite bitmap.
  fn sprite_half_w() : Float do 16.0 end
  fn sprite_half_h() : Float do 24.0 end

  -- One hull point: a (grid col, grid row) pair, scaled to sprite-space,
  -- expanded by shield_scale() from the sprite's center, then re-offset so
  -- (0, 0) sprite-center maps to (0, 0) here -- the same local frame
  -- draw_ball's Some(sprite) branch already draws the sprite bitmap in.
  pfn hull_point(col : Int, row : Int) : (Float, Float) do
    let sx = int_to_float(col) * sprite_col_scale()
    let sy = int_to_float(row) * sprite_row_scale()
    let ex = sprite_half_w() + (sx - sprite_half_w()) * shield_scale()
    let ey = sprite_half_h() + (sy - sprite_half_h()) * shield_scale()
    (ex - sprite_half_w(), ey - sprite_half_h())
  end

  -- Right-edge points use right + 1 (the outer edge of the rightmost filled
  -- cell), matching draw_sprite_run's end_col+1 convention, so the hull
  -- sits at the pixel boundary rather than one cell short.
  pfn right_edge_points(extents : List(SpriteRowExtent)) : List((Float, Float)) do
    match extents do
      Nil -> Nil
      Cons(e, rest) -> Cons(hull_point(e.right + 1, e.row), right_edge_points(rest))
    end
  end

  pfn left_edge_points(extents : List(SpriteRowExtent)) : List((Float, Float)) do
    match extents do
      Nil -> Nil
      Cons(e, rest) -> Cons(hull_point(e.left, e.row), left_edge_points(rest))
    end
  end

  doc "The player-ship shield's outline: the right edge walked top-to-bottom, then the left edge walked bottom-to-top, closing into a hull polygon that traces the ship's actual silhouette (mast prongs, forked legs) rather than a generic bounding shape, scaled up slightly from the sprite itself. Computed once at boot (see fx_init) and reused every frame the shield is active -- pure, no Canvas calls."
  pfn shield_hull_points() : List((Float, Float)) do
    let bottom_to_top = sprite_row_extents()
    let top_to_bottom = List.reverse(bottom_to_top)
    List.append(right_edge_points(top_to_bottom), left_edge_points(bottom_to_top))
  end

  -- Walks a point list, calling Canvas.move_to for the first point and
  -- Canvas.line_to for the rest -- must be called between a Canvas.begin_path
  -- and a Canvas.close_path/Canvas.stroke, same pattern as draw_asteroid_edges.
  pfn draw_hull_points(ctx, points : List((Float, Float)), first : Bool) : Unit do
    match points do
      Nil -> ()
      Cons((px, py), rest) ->
        if first do
          Canvas.move_to(ctx, px, py)
        else
          Canvas.line_to(ctx, px, py)
        end
        draw_hull_points(ctx, rest, false)
    end
  end

```
